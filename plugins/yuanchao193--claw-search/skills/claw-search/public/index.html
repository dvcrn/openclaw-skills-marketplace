简单易用

几行代码即可集成到你的 AI 应用中

JavaScript

const response = await fetch('https://api.claw-search.com/api/search', {
  method: 'POST',
  headers: { 'Content-Type': 'application/json' },
  body: JSON.stringify({
    query: 'openclaw ai agent',
    count: 10
  })
});
const data = await response.json();

Python

import requests

response = requests.post(
    'https://api.claw-search.com/api/search',
    json={
        'query': 'openclaw ai agent',
        'count': 10
    }
)
data = response.json()
